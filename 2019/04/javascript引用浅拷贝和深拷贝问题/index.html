JavaScript引用、浅拷贝和深拷贝问题

四月 2, 2019 发布在 javascript

定义两个对象作为实验的例子

var person = {
    name: "zhangsan",
    age: 22,
    hobbies: {
        game: "LOL",
        sport: "basketball"
    }
}

var girl = {
    name: "krystal",
    age: 24,
    height: 170
}

引用

var person1 = person;
var girl1 = girl;

赋值之后，相当于person1和person指向同一个对象，girl1和girl指向同一个对象。

person1.name = "lisi"
girls.age = 18
console.log(person.name);  //lisi
console.log(girl.age);     //18

显然，因为指向同一个对象，修改person1和girl1的属性值，person和girl的属性值也会随之改变。

浅拷贝

注意：浅拷贝不是引用

浅拷贝是将一个对象的所有属性克隆给一个新的对象

//浅拷贝
function shallowCopy(obj) {
    var result = Array.isArray(obj) ? [] : {};
    for (var prop in obj) {
        if (obj.hasOwnProperty(prop)) {
            result[prop] = obj[prop];
        }
    }
    return result;
}

仔细观察该函数，可以发现，浅拷贝只是简单的将对象的第一层拷贝出来给一个新的对象。但是如果原对象某一属性是引用类型，那么就会出现直接引用、赋值的情况。以上面定义的person对象为例：

var person1 = shallowCopy(person);
person1.hobbies.sport = "football";
console.log(person.hobbies.sport);  //输出football

修改新对象的值依然会影响原对象，这显然与我们的需求不符合，这时候深拷贝就派上用场了！

深拷贝

在搞明白浅拷贝的缺陷之后，我们很容易实现深拷贝。既然浅拷贝只拷贝了第一层属性，那么深拷贝就可以判断对象的属性，如果是引用类型，就递归调用，直到不是引用类型为止。

//深拷贝
function deepCopy(obj) {
    var result = Array.isArray(obj) ? [] : {};
    if (obj && typeof(obj) === "object") {
        for (var prop in obj) {
            if (obj.hasOwnProperty(prop)) {
                if (obj[prop] && typeof(obj[prop]) === "object") {
                    result[prop] = deepCopy(obj[prop]);
                } else {
                    result[prop] = obj[prop];
                }
            }
        }
    }
    return result;
}

深拷贝方法2：利用JSON函数，做两次转换：

function deepCopy2(obj) {
    var result = JSON.stringify(obj);
    result = JSON.parse(result);
    return result;
}

测试用例

var person = {
    name: "zhangsan",
    age: 22
}

var arr1 = ["111", "222", "333", person];
var arr2 = arr1;
var arr3 = shallowCopy(arr1);
var arr4 = deepCopy(arr1);

console.log(arr1);
console.log(arr2);
console.log(arr3);
console.log(arr4);

arr4[3].age = 44;
console.log("------------------------------------------------------")
console.log(arr1);
console.log(arr2);
console.log(arr3);
console.log(arr4);
